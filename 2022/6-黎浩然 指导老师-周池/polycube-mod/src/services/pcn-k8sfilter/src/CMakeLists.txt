include(${PROJECT_SOURCE_DIR}/cmake/LoadFileAsVariable.cmake)

aux_source_directory(serializer SERIALIZER_SOURCES)
aux_source_directory(api API_SOURCES)
aux_source_directory(default-src SRC_SOURCES)

include_directories(serializer)
include_directories(interface)
include_directories(default-src)

# Needed to load files as variables
include_directories(${CMAKE_CURRENT_BINARY_DIR})

add_library(pcn-k8sfilter SHARED
  ${SERIALIZER_SOURCES}
  ${API_SOURCES}
  ${SRC_SOURCES}
  K8sfilter.cpp
  Ports.cpp
  K8sfilter-lib.cpp)

# load ebpf datapath code in std::string variables
load_file_as_variable(pcn-k8sfilter K8sfilter_dp.c k8sfilter_code)

# load datamodel in a variable
load_file_as_variable(pcn-k8sfilter ../datamodel/k8sfilter.yang k8sfilter_datamodel)

target_link_libraries(pcn-k8sfilter
  polycube
  uuid)

# Specify shared library install directory

set(CMAKE_INSTALL_LIBDIR /usr/lib)

install(
    TARGETS
    pcn-k8sfilter
    DESTINATION
    "${CMAKE_INSTALL_LIBDIR}"
)
